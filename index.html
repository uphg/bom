# Carob 文档

# 元素

# appendappend(el, nodes[, ...args])

在当前元素的子元素后追加元素

append(document.body, [el1, el2])
append(document.body, el1, el2, ...elN)

# prependprepend(el, nodes[, ...args])

在当前元素的子元素前追加元素

prepend(document.body, [el1, el2])
prepend(document.body, el1, el2, ...elN)

# removeChildrenremoveChildren(el)

删除当前元素下所有子节点

removeChildren(el)

# toElementtoElement(innerHTML, [children])

将字符串的 HTML 内容转为 DOM 元素

const app = toElement(`
  <div class="app">
    <button id="button">Hi</button>
  </div>
`)

append(document.body, app)

# getScrollbarWidthgetScrollbarWidth()

获取当前页面滚动条宽度

getScrollbarWidth() // => 17px

# getScrollParentgetScrollParent(el)

获取当前元素上层的第一个滚动父节点

getScrollParent(el) // document.body

# onon(el, eventName, selectorOrHandler, [handlerOrOptions], [options])

对元素进行事件监听/事件委托

const app = toElement(`
  <div class="app">
    <button id="button">我是按钮</button>
  </div>
`)

const button = app.querySelector('#button')

const onClick = (e) => {
  console.log(e)
}

on(app, 'click', '#button', onClick)
on(button, 'click', onClick)

# offoff(el, eventName, handler, [options])

取消事件监听/事件委托

off(el, 'click', onClick)

# addClassaddClass(el, className[, ...args])

给元素添加 class 类名

addClass(el, 'blue')
addClass(el, 'a1 a2')
addClass(el, 'b1', 'b2')

el.getAttribute('class')
// => blue a1 a2 b1 b2

# removeClassremoveClass(el, className[, ...args])

给元素删除 class 类名

el.classList.add('red', 'blue', 'a1', 'a2', 'b1', 'b2')

removeClass(el, 'red')
removeClass(el, 'a1 a2')
removeClass(el, 'b2', 'b3')

el.getAttribute('class')
// => blue

# hasClasshasClass(el, className)

判断元素是否存在 class 类名

el.classList.add('red', 'blue', 'green')

hasClass(el, 'red')
// => true

# getStylegetStyle(el, styleName)

获取元素 style 样式

el.style.height = '60px'

getStyle(el, 'height')
// => 60px

# setStylesetStyle(el, styles)

给元素设置 style 样式

setStyle(el, {
  marginTop: '20px',
  'margin-bottom': '20px'
})
setStyle(el, 'paddingTop', '10px')
setStyle(el, 'line-height', '1.5')

el.getAttribute('style')
// => margin-top: 20px; margin-bottom: 20px; padding-top: 10px; line-height: 1.5;

# getIndexgetIndex(el)

获取当前元素在父元素 children 中的索引，如果元素为空或不存在父元素，会返回 -1

getIndex(document.body) // 1
getIndex(null) // -1
